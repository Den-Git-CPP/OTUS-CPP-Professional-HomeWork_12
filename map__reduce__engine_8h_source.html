133 std::vector<map_result_t> merge_map_results(int i, int j) {
134 assert(i <= j);
135 if (i == j){
136 return map_results[i];
137 }
138 std::vector<map_result_t> result1;
139 std::vector<map_result_t> result2;
140 if (i + 1 == j) {
141 result1 = map_results[i];
142 result2 = map_results[j];
143 } else {
144 int middle = (i + j) / 2;
145 std::future<std::vector<map_result_t>> result1_future = std::async(
146 std::launch::async,
147 [this, i, middle] {
148 return this->merge_map_results(i, middle);
149 });
150 result2 = merge_map_results(middle + 1, j);
151 result1 = result1_future.get();
152 }
153
154 std::vector<map_result_t> result{};
155 result.reserve(result1.size() + result2.size());
156
157 size_t i1 = 0;
158 size_t i2 = 0;
159 while (i1 < result1.size() || i2 < result2.size()) {
160 if (i1 >= result1.size()){
161 result.emplace_back(result2[i2++]);
162 }else if (i2 >= result2.size()){
163 result.emplace_back(result1[i1++]);
164 }else if (result1[i1].first <= result2[i2].first){
165 result.emplace_back(result1[i1++]);
166 }else{
167 result.emplace_back(result2[i2++]);
168 }
169 }
170 return result;
